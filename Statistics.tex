\begin{definition}[1-d Gassian]
A random variable $X\sim N(\mu,\sigma^2)$,
\begin{equation}
N(\mu,\sigma^2)(x) = \frac{1}{\sqrt{2\pi}\sigma^2}\exp\left\{-\frac{(x-\mu)^2}{2\sigma^2}\right\},
\end{equation}
where $\mu, \sigma$ are the mean and variance respectively, called the Gassisan random variable. Furthermore, it is called normal random variable if $X\sim N(0,1)$.
\end{definition}

\begin{definition}[n-d Gassian]
A random varialbe $Y$ is called n-dimensional Gassian random variable if $\boldsymbol{X}=(X_1,X_2,...,X_n)$, where $X_i\iid\sim N(\mu_i,\sigma_i^2)$.
\end{definition}
